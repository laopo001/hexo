    fn main() {
            #[derive(Debug)]
            struct A(i32);
            let mut a = A(2);
            let p_a: *mut A = &mut a;
       
            unsafe {
                (*p_a).0 = 4;
                // println!("A:{:?}", *p_a);
                let q = *p_a;
                println!("A:{:?}", q);
            }
            println!("A:{:?}", a);
    }
    
    

    #[derive(Debug)]
    struct A(i32,Box<B>);
    #[derive(Debug)]
    struct B(i32);
    fn mut1(a:&mut A){
        a.0=123;
        mut2(a.1.as_mut());
        a.0=1237;
    }
    
    fn mut2(a:&mut B){
        (*a).0=1253;
    }
    
    fn main() {
        let mut a = A(2,Box::new(B(2)));
        mut1(&mut a);
        println!("{:?}",a);
    }
